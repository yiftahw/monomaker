30def get_all_branches(repo_path: str, verbose: bool = False, raw: bool = False) -> List[str]: 

31 cmd = "git branch -a" 

32 out = exec_cmd(cmd, cwd=repo_path) 

33 branches = set() 

34 if verbose: 

35 print(f"Branches in repo {repo_path}:\n{out.stdout}") 

36 if raw: 

37 return [line.strip() for line in out.stdout.splitlines()] 

38 for line in out.stdout.splitlines(): 

39 line = line.strip() 

40 if line.startswith("*"): 

41 line = line[1:].strip() 

42 line = line.strip() 

43 if line.find("HEAD") != -1 or line.find("no branch") != -1: 

44 continue 

45 if line.startswith("remotes/origin/"): 

46 line = line[len("remotes/origin/"):] 

47 branches.add(line) 

48 return list(branches)

50def get_head_branch(repo_path: str) -> Optional[str]: 

51 """ 

52 Returns the default branch of the given repo, or None if it cannot be determined. 

53 """ 

54 cmd = "git rev-parse --abbrev-ref HEAD" 

55 out = exec_cmd(cmd, cwd=repo_path) 

56 default_branch = out.stdout.strip() 

57 if default_branch == "HEAD" or default_branch.find("no branch") != -1 or default_branch == "": 

58 return None 

59 return default_branch

62def get_all_submodules(repo_path: str) -> List[SubmoduleDef]: 

63 """ 

64 Returns list of submodule paths in the given repo (at its current HEAD) 

65 """ 

66 # retrieve all submodule commit hashes 

67 # git submodule status is not recursive, which is good for us 

68 submodule_hashes_raw = exec_cmd("git submodule status", cwd=repo_path).stdout 

69 submodule_hashes = dict() 

70 for line in submodule_hashes_raw.splitlines(): 

71 line = line.strip() 

72 if line == "": 

73 continue 

74 parts = line.split() 

75 if len(parts) < 2: 

76 continue 

77 commit_hash = parts[0].lstrip("-+") 

78 path = parts[1] 

79 submodule_hashes[path] = commit_hash 

80 

81 # read .gitmodules file to get the submodule paths and URLs 

82 gitmodules_path = os.path.join(repo_path, ".gitmodules") 

83 if not os.path.isfile(gitmodules_path): 

84 return [] 

85 config = configparser.ConfigParser() 

86 config.read(gitmodules_path) 

87 submodules = [] 

88 for section in config.sections(): 

89 if section.startswith("submodule "): 

90 path = config[section].get("path") 

91 if not path: 

92 continue 

93 url = config[section].get("url") 

94 if not url: 

95 continue 

96 commit_hash = submodule_hashes.get(path, "") 

97 if commit_hash == "": 

98 raise RuntimeError(f"Cannot find commit hash for submodule at path {path}") 

99 submodules.append(SubmoduleDef(path, url, commit_hash)) 

100 return submodules

103def import_meta_repo(monorepo_root_dir: str, metarepo_root_dir: str): 

104 """ 

105 It is expected that both folders are git repositories, and that the metarepo 

106 is already cloned locally. 

107 """ 

108 metarepo_branches = get_all_branches(metarepo_root_dir)

112 for branch in metarepo_branches: 

113 print(f"=== Importing meta:{branch} ===")

116 # cleanup 

117 exec_cmd(f"git remote remove metarepo", cwd=monorepo_root_dir)

132class SubmoduleImportInfoEntry: 

133 """Represents which submodule branch applied to which metarepo branch.""" 

134 monorepo_branch: str # branch name in the new monorepo 

135 metarepo_branch: str # what branch was used to import the metarepo files 

136 submodule_branch: str # what branch was used to import the submodule files 

137 submodule_nested_submodules: List[SubmoduleDef] # nested submodules in this submodule branch 

138 def __eq__(self, other): 

139 if not isinstance(other, SubmoduleImportInfoEntry): 

140 print("Other is not SubmoduleImportInfoEntry") 

141 return False 

142 return (self.monorepo_branch == other.monorepo_branch and 

143 self.metarepo_branch == other.metarepo_branch and 

144 sorted(self.submodule_nested_submodules) == sorted(other.submodule_nested_submodules)) 

145 def __str__(self): 

146 return f"SubmoduleImportInfoEntry(monorepo_branch={self.monorepo_branch}, metarepo_branch={self.metarepo_branch}, submodule_branch={self.submodule_branch}, nested_submodules={self.submodule_nested_submodules})"

148class SubmoduleImportInfo: 

149 submodule_relative_path: str 

150 entries: List[SubmoduleImportInfoEntry] 

151 def __init__(self, submodule_relative_path: str): 

152 self.submodule_relative_path = submodule_relative_path 

153 self.entries = [] 

154 def add_entry(self, monorepo_branch: str, metarepo_branch: str, submodule_branch: str, nested_submodules: Optional[List[SubmoduleDef]] = None): 

155 self.entries.append(SubmoduleImportInfoEntry(monorepo_branch, metarepo_branch, submodule_branch, nested_submodules or [])) 

156 def __str__(self): 

157 s = f"Submodule Import Info for {self.submodule_relative_path}:\n" 

158 

159 for entry in self.entries: 

160 s += f" - {entry.monorepo_branch}: metarepo branch: {entry.metarepo_branch}, submodule branch: {entry.submodule_branch}\n" 

161 s += f" - nested submodules:\n" if len(entry.submodule_nested_submodules) > 0 else "" 

162 for nested in entry.submodule_nested_submodules: 

163 s += f" path: {nested.path}, url: {nested.url}, commit: {nested.commit_hash}\n" 

164 return s 

165 def __eq__(self, other): 

166 if not isinstance(other, SubmoduleImportInfo): 

167 print("Other is not SubmoduleImportInfo") 

168 return False 

169 if self.submodule_relative_path != other.submodule_relative_path: 

170 print(f"Submodule paths differ: {self.submodule_relative_path} != {other.submodule_relative_path}") 

171 return False 

172 if len(self.entries) != len(other.entries): 

173 print(f"Number of entries differ: {len(self.entries)} != {len(other.entries)}") 

174 return False 

175 # sort each list prior to comparison 

176 def key_func(e: SubmoduleImportInfoEntry): 

177 return (e.metarepo_branch, e.submodule_branch) 

178 self.entries.sort(key=key_func) 

179 other.entries.sort(key=key_func) 

180 for e1, e2 in zip(self.entries, other.entries): 

181 if e1 != e2: 

182 print(f"Entries differ:\n{e1}\n{e2}") 

183 return False 

184 return True

210def import_submodule(monorepo_root_dir: str, 

211 submodule_repo_url: str, 

212 submodule_path: str, 

213 metarepo_default_branch: str, 

214 metarepo_branches: Set[str], 

215 expected_branches: Optional[Set[str]] = None, 

216 metarepo_tracked_branches: Optional[Set[str]] = None) -> SubmoduleImportInfo: 

217 """ 

218 It is expected that monorepo_root_dir points to a git repository where the submodule will be imported. 

219 the submodule will be cloned from submodule_repo_url, and all its branches will be imported under submodule_path in the monorepo. 

220 

221 metarepo_tracked_branches: all branches in the metarepo that track this submodule. 

222 """ 

223 report = SubmoduleImportInfo(submodule_path) 

224 with tempfile.TemporaryDirectory() as tempdir: 

225 # First, make a minimal clone just to get branch information 

226 info_clone_dir = os.path.join(tempdir, "info_clone") 

227 print(f"Cloning repository to discover branches...") 

228 exec_cmd(f"git clone {submodule_repo_url} {info_clone_dir}") 

229 exec_cmd("git fetch --all --prune", cwd=info_clone_dir) 

230 

231 # Get the default branch (after cloning, HEAD points to the default branch) 

232 submodule_default_branch = get_head_branch(info_clone_dir) 

233 

234 # Get all branches from the fresh clone 

235 submodule_branches = set(get_all_branches(info_clone_dir)) 

236 print(f"Found branches for submodule {submodule_path}: {submodule_branches}") 

237 if expected_branches is not None and submodule_branches != expected_branches: 

238 raise RuntimeError(f"Submodule branches mismatch. Expected: {expected_branches}, Found: {submodule_branches}") 

239 

240 # Process each branch 

241 branches_dir = os.path.join(tempdir, "branches") 

242 ensure_dir(branches_dir) 

243 

244 if metarepo_tracked_branches is None: 

245 print(f"import_submodule(): warning: metarepo_tracked_branches is None, all submodule branches will be imported into the monorepo.") 

246 

247 # branches_clusure: all branches that need to be considered for this submodule 

248 # see comments below for details 

249 branches_closure = submodule_branches.copy() 

250 if metarepo_tracked_branches is not None: 

251 branches_closure.update(metarepo_tracked_branches) 

252 

253 for branch in branches_closure: 

254 # for each {metarepo/branch}, if: 

255 # 1. branch exists in metarepo but doesn't track submodule -> skip importing submodule branch into it 

256 # 2. branch exists in metarepo, exists in submodule -> import submodule branch into it (if submodule is tracked in the metarepo branch) 

257 # 3. branch exists in metarepo, doesn't exist in submodule -> import the submodule's default branch (if needed, see above) 

258 # 4. branch doesn't exist in metarepo, exists in submodule -> branch out from metarepo's default branch, import the submodule (if needed, see above) 

259 

260 # NOTE: in case 4, it is assumed that the metarepo itself was already imported into the monorepo, 

261 # so the default branch should exist fully in the monorepo, and it is safe to branch out from it. 

262 

263 # case 1: branch exists in metarepo but doesn't track submodule, or doesn't exist and default branch doesn't track it either 

264 # -> skip importing submodule branch into it 

265 metarepo_default_branch_tracks_submodule = (metarepo_tracked_branches is not None and metarepo_default_branch in metarepo_tracked_branches) 

266 branch_exists_in_metarepo = branch in metarepo_branches 

267 submodule_not_tracked_in_metarepo_branch = metarepo_tracked_branches is not None and branch not in metarepo_tracked_branches 

268 if (branch_exists_in_metarepo and submodule_not_tracked_in_metarepo_branch) or (not branch_exists_in_metarepo and not metarepo_default_branch_tracks_submodule): 

269 print(f"Skipping import of submodule {submodule_path} branch {branch} into monorepo, as metarepo branch does not track this submodule.") 

270 print(f"Details:") 

271 print(f"branch_exists_in_metarepo: {branch_exists_in_metarepo}") 

272 print(f"submodule_not_tracked_in_metarepo_branch: {submodule_not_tracked_in_metarepo_branch}") 

273 print(f"metarepo_default_branch_tracks_submodule: {metarepo_default_branch_tracks_submodule}") 

274 print(f"") 

275 print(f"metarepo_tracked_branches: {metarepo_tracked_branches}") 

276 print(f"metarepo default_branch: {metarepo_default_branch}") 

277 continue 

278 

279 # if some submodule doesn't contain a feature branch that DOES exist in the metarepo, 

280 # we should import the submodule's default branch into the metarepo's feature branch 

281 branch_to_import = branch 

282 if branch not in submodule_branches: 

283 if submodule_default_branch is None or submodule_default_branch not in submodule_branches: 

284 raise RuntimeError(f"Cannot import submodule branch {branch} into monorepo, as it does not exist in the submodule, and default branch cannot be determined.") 

285 print(f"Submodule branch {branch} does not exist in submodule, using default branch {submodule_default_branch} instead.") 

286 branch_to_import = submodule_default_branch 

287 

288 # prepare monorepo branch 

289 metarepo_branch_used = branch 

290 if not branch_exists_in_metarepo: 

291 # first switch to the default branch 

292 exec_cmd(f"git switch {metarepo_default_branch}", cwd=monorepo_root_dir) 

293 # then create the new branch from it 

294 exec_cmd(f"git switch -c {branch}", cwd=monorepo_root_dir) 

295 print(f"Created new monorepo branch {branch} from metarepo default branch {metarepo_default_branch} to import submodule into it.") 

296 metarepo_branch_used = metarepo_default_branch 

297 else: 

298 # branch exists in metarepo, so we should switch to it 

299 exec_cmd(f"git switch {branch}", cwd=monorepo_root_dir) 

300 print(header_string(f"Importing {submodule_path}:{branch_to_import} to monorepo:{branch}")) 

301 

302 # prepare submodule branch clone (isolated workspace, git-filter-repo modifies its git history) 

303 branch_clone_dir_name = f"clone_{branch_to_import.replace('/', '_')}" 

304 branch_clone_dir = os.path.join(branches_dir, branch_clone_dir_name) 

305 exec_cmd(f"rm -rf {branch_clone_dir}") # might already exist if multiple metarepo branches point to same submodule branch 

306 exec_cmd(f"git clone -b {branch_to_import} --single-branch {submodule_repo_url} {branch_clone_dir}") 

307 

308 # Record in report 

309 nested_submodules = get_all_submodules(branch_clone_dir) 

310 report.add_entry(branch, metarepo_branch_used, branch_to_import, nested_submodules) 

311 

312 # Run filter-repo on the isolated clone to move everything under submodule_path 

313 exec_cmd(f"python3 {GIT_FILTER_REPO} --force --to-subdirectory-filter {submodule_path}", cwd=branch_clone_dir) 

314 

315 # need to remove all the files in the monorepo that are under submodule_path 

316 submodule_full_path_in_monorepo = os.path.join(monorepo_root_dir, submodule_path) 

317 if os.path.exists(submodule_full_path_in_monorepo): 

318 print(f"Removing existing files in monorepo at {submodule_full_path_in_monorepo} ...") 

319 exec_cmd(f"git rm -rf {submodule_path}", cwd=monorepo_root_dir) 

320 exec_cmd(f"git commit -m 'Remove existing submodule path {submodule_path} before merge'", cwd=monorepo_root_dir) 

321 else: 

322 print(f"No existing files to remove in monorepo at {submodule_full_path_in_monorepo}.") 

323 

324 # Add the filtered clone as a temporary remote, and merge its branch into the monorepo branch 

325 branch_clone_abs = os.path.abspath(branch_clone_dir) 

326 remote_name = f"tmp_{branch.replace('/', '_').replace('-', '_')}" 

327 

328 exec_cmd(f"git remote add {remote_name} {branch_clone_abs}", cwd=monorepo_root_dir) 

329 exec_cmd(f"git fetch {remote_name}", cwd=monorepo_root_dir) 

330 exec_cmd(f"git merge {remote_name}/{branch_to_import} --allow-unrelated-histories -m 'Merge submodule {submodule_path} branch {branch_to_import} to monorepo branch {branch}'", cwd=monorepo_root_dir) 

331 

332 # Cleanup remote (the clone directory will be cleaned up by tempdir) 

333 exec_cmd(f"git remote remove {remote_name}", cwd=monorepo_root_dir) 

334 

335 # if we found any nested submodules in this submodule, we need to remove `submodule_path/.gitmodules` file from the monorepo 

336 # and then register the actual nested submodules in the monorepo 

337 for nested_submodule in nested_submodules: 

338 nested_submodule_relative_path_in_monorepo = os.path.join(submodule_path, nested_submodule.path) 

339 nested_submodule_abs_path = os.path.join(monorepo_root_dir, nested_submodule_relative_path_in_monorepo) 

340 print(header_string(f"Registering nested submodule {nested_submodule_relative_path_in_monorepo} in monorepo branch {branch}")) 

341 # remove .gitmodules file if it exists 

342 gitmodules_in_monorepo = os.path.join(monorepo_root_dir, submodule_path, ".gitmodules") 

343 if os.path.isfile(gitmodules_in_monorepo): # if we have more than one nested submodule in the same subdirectory, we only need to remove it once 

344 print(f"Removing .gitmodules file for nested submodules at {gitmodules_in_monorepo} ...") 

345 exec_cmd(f"git rm {os.path.join(submodule_path, '.gitmodules')}", cwd=monorepo_root_dir) 

346 exec_cmd(f"git commit -m 'Remove .gitmodules file for nested submodules in {submodule_path}'", cwd=monorepo_root_dir) 

347 # remove nested submodule entry from subdirectory 

348 nested_submodule_exists = os.path.exists(nested_submodule_abs_path) 

349 if nested_submodule_exists: 

350 print(f"Removing nested submodule files at {nested_submodule_abs_path} ...") 

351 exec_cmd(f"git rm -rf {nested_submodule_relative_path_in_monorepo}", cwd=monorepo_root_dir) 

352 exec_cmd(f"git commit -m 'Remove nested submodule {nested_submodule_relative_path_in_monorepo} files before tracking it in monorepo'", cwd=monorepo_root_dir) 

353 # re-register nested submodule in monorepo 

354 # `--force` is needed in case multiple branches contain the same nested submodule (likely) 

355 commit_hash = nested_submodule.commit_hash 

356 exec_cmd(f"git submodule add --force {nested_submodule.url} {nested_submodule_relative_path_in_monorepo}", cwd=monorepo_root_dir) 

357 exec_cmd(f"git checkout {commit_hash}", cwd=nested_submodule_abs_path) 

358 exec_cmd(f"git commit -m 'Add nested submodule {nested_submodule_relative_path_in_monorepo} at commit {commit_hash}'", cwd=monorepo_root_dir) 

359 

360 return report

362def get_metarepo_tracked_submodules_mapping(repo_path: str) -> Mapping[SubmoduleDef, Set[str]]: 

363 """ 

364 Scans all branches in the given metarepo,  

365 returns a mapping of {submodule -> set of branches that track them in the metarepo}. 

366 """ 

367 tracked_submodules: Mapping[SubmoduleDef, Set[str]] = dict() 

368 branches = get_all_branches(repo_path) 

369 for branch in branches: 

370 print(f"--- Scanning branch {branch} for submodules ---") 

371 exec_cmd(f"git checkout {branch}", cwd=repo_path) 

372 submodules_in_branch = get_all_submodules(repo_path) 

373 for submodule in submodules_in_branch: 

374 # ignore commit hash for tracking purposes,  

375 # as we only care here about existence of submodule in branch 

376 # TODO: find a cleaner solution for this? 

377 submodule.commit_hash = "" 

378 if submodule not in tracked_submodules: 

379 tracked_submodules[submodule] = set() 

380 tracked_submodules[submodule].add(branch) 

381 return tracked_submodules

423def main_flow(params: WorkspaceMetadata) -> MigrationReport: 

424 report = MigrationReport() 

425 

426 # destructure params 

427 monorepo_root_dir = params.monorepo_root_dir 

428 metarepo_root_dir = params.metarepo_root_dir 

429 metarepo_default_branch = params.metarepo_default_branch 

430 metarepo_branches = params.metarepo_branches 

431 

432 # Import metarepo 

433 import_meta_repo(monorepo_root_dir, metarepo_root_dir) 

434 

435 # Some branches in the metarepo may or may not track some submodules 

436 # So we need to scan all metarepo branches for submodules, to know which ones to import. 

437 # for each submodule, we will bookkeep which metarepo branches track it. 

438 metarepo_tracked_submodules_mapping = get_metarepo_tracked_submodules_mapping(metarepo_root_dir) 

439 

440 print(f"Submodules to import:") 

441 for submodule in metarepo_tracked_submodules_mapping: 

442 print(f"path: {submodule.path}\nurl: {submodule.url}\n") 

443 

444 # for each submodule, we will do a fresh clone, and then process it 

445 for submodule in metarepo_tracked_submodules_mapping:

447 report.add_submodule_entry(submodule.path, submodule_report) 

448 

449 print(header_string("Merge Complete")) 

450 return report
